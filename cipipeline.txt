
node {
   def mvnHome
   stage('Preparation') {
      git 'https://github.com/AdityaSP/MyAppDemo'
      mvnHome = tool 'mvn'
   }
   stage('Build') {
         sh "'${mvnHome}/bin/mvn' -Dmaven.test.failure.ignore clean package"
   }
      stage('CI DB Deploy'){
       sh "mysql -uroot -proot < '/var/lib/jenkins/workspace/cicddemo/dbrelease/databaserelease.sql'"
       sh "echo Finished with database update"
   }
   
    stage('CI WAR Deploy'){
       sh "echo Deploy War"
       sh "cp '/var/lib/jenkins/workspace/cicddemo/target/MyAppDemo.war' '/var/lib/tomcat7/webapps/'"
       sh "echo Finished Deploying WAR"
   }
   stage('CI Results') {
      junit '**/target/surefire-reports/TEST-*.xml'
      archive 'target/*.war'
   }
   
   stage('CD Preparation') {
      sh "cp '/var/lib/jenkins/workspace/cicddemo/target/MyAppDemo.war' '/home/vagrant/chefdemo/cookbooks/devopsdemo/files/'"
      sh "cp '/var/lib/jenkins/workspace/cicddemo/dbrelease/databaserelease.sql' '/home/vagrant/chefdemo/cookbooks/devopsdemo/files/'"
   }
   stage('CD Knife Runs'){
       sh "cd /home/vagrant/chefdemo;knife cookbook upload devopsdemo"
   }
   stage('CD Node Runs'){
       sh "cd /home/vagrant/chefdemo;echo vagrant | knife ssh 10.1.1.34 'sudo chef-client' --manual-list --ssh-user vagrant --identity-file /vagrant/vagrant_private_key"
   }
   
 

}
